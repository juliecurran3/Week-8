SELECT sum(salary) Salary_Sum, departments.dept_name, salaries.from_date, salaries.to_date     
FROM salaries    
INNER JOIN dept_emp ON dept_emp.emp_no = salaries.emp_no     
INNER JOIN departments ON departments.dept_no = dept_emp.dept_no  
GROUP BY dept_name 
HAVING dept_name = 'Marketing' 
and salaries.from_date >= '1990-01-01' and salaries.to_date <= '1992-12-31';
